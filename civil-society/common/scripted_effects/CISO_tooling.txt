#==========TOOLING==========#
ciso_civsoc_process_tooling_handle_creation = {
	if = {
		limit = {
			$ci$_creation_trigger = yes
		}
		add_to_variable_list = {
			name = ciso_civil_institutions
			target = flag:$ci$
		}
	}
	else = {
		remove_list_variable = {
			name = ciso_civil_institutions
			target = flag:$ci$
		}
	} 
}

ciso_init_civsoc = {
	every_scope_state = {
		limit = {
			ciso_state_has_civil_society = yes
		}
		ciso_civsoc_process_monthly = yes
		# do something
	}
}
