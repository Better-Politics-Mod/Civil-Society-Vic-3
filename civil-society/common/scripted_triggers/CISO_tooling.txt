ciso_state_has_civil_society = {
	OR = {
		OWNER = {
			total_population > 10000000
		}
		AND = {
			is_capital = yes
			OWNER = { OR = {
				is_player = yes
				country_rank >= rank_value:insignificant_power
			} }
		}
		OWNER = {
			country_rank >= rank_value:great_power
		}
	}
}

ciso_ci_is_radical = {
	$ci$_is_radical = yes
}
