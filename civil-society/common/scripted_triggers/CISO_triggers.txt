bpm_state_has_civil_society = {
	OR = {
		OWNER = {
			total_population > 10000000
		}
		is_capital = yes
		OWNER = {
			country_rank >= rank_value:great_power
		}
	}
}


#======BPMCS CIVIL INSTITUTION CREATION TRIGGERS======#
# REGISTER CIVINST 4

bpmcs_ci_manufacturing_corporations_creation_trigger = {
	custom_tooltip = {
		text = "bpmcs_ci_manufacturing_corporations_creation_trigger_manufacturing"
		any_scope_building = {
			level > 10
			is_building_group = bg_manufacturing
			private_ownership_fraction > 0.2
		}
	}
	custom_tooltip = {
		text = "bpmcs_ci_manufacturing_corporations_creation_trigger_capitalist_ratio"
		pop_type_percent_state = {
			target = pop_type:capitalists
			value >= 0.0025
		}
	}
}

bpmcs_ci_heavy_industry_unions_creation_trigger = {
	custom_tooltip = {
		text = "bpmcs_ci_heavy_industry_unions_creation_trigger_heavy_industry"
		any_scope_building = {
			level > 5
			is_building_group = bg_heavy_industry
		}
	}
	custom_tooltip = {
		text = "bpmcs_ci_heavy_industry_unions_creation_trigger_machinists"
		pop_type_percent_state = {
			target = pop_type:machinists
			value >= 0.025
		}
	}
}

bpmcs_ci_royal_court_creation_trigger = {
	custom_tooltip = {
		text = "bpmcs_ci_royal_court_creation_trigger_is_royal_capital"
		is_capital = yes
		owner = { 
			has_law = law_type:law_monarchy
		}
	}
}

	}
}
