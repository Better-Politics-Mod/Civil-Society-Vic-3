on_monthly_pulse_state = {
	on_actions = {
		ciso_on_monthly
	}
}

on_half_yearly_pulse_state = {
	on_actions = {
		ciso_on_halfyearly
	}
}

on_game_started = {
	on_actions = {
		ciso_on_game_started
	}
}

ciso_on_monthly = {
	effect = {
		if = {
			limit = {
				ciso_state_has_civil_society = yes
			}
			ciso_civsoc_process_monthly = yes
		}
	}
}

ciso_on_halfyearly = {
	effect = {
		if = {
			limit = {
				ciso_state_has_civil_society = yes
			}
			ciso_civsoc_process_halfyearly = yes
		}
	}
}

ciso_on_game_started = {
	effect = {
		ciso_init_civsoc_global = yes
		every_country = {
			ciso_init_civsoc = yes
		}


		add_to_global_variable_list = {
			name = ciso_society_measures
			target = flag:ciso_measure_test_measure
		}
	}
}
