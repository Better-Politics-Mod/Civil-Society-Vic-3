ciso_ci_heavy_industry_unions = {
	icon = "gfx/interface/icons/civil_institution_icons/heavy_industry_unions.dds"

	visible = {
		always = yes
	}

	possible = {
		custom_tooltip = {
			text = "ciso_ci_heavy_industry_unions_creation_trigger_heavy_industry"
			any_scope_building = {
				level > 5
				is_building_group = bg_heavy_industry
			}
		}
		custom_tooltip = {
			text = "ciso_ci_heavy_industry_unions_creation_trigger_machinists"
			pop_type_percent_state = {
				target = pop_type:machinists
				value >= 0.025
			}
		}
	}

	measure_weights = {
		value = 0
		if = {
			limit = {
				scope:measure = flag:ciso_ms_union_hall
			}
			add = 1
		}
	}

	is_radical = {
		OWNER = {
			has_law = law_type:law_combination_acts
		}
	}

	is_loyalist = {
		OWNER = {
			OR = {
				has_law = law_type:law_corporatized_unions
				has_law = law_type:law_factory_councils
			}
		}
	}

	# stance 
	# 0 = defensive
	# >0 = aggressive
	# <0 = cooperative
	stance = {
		value = 0
	}

	social_impact = {
		value = organization
		multiply = size # the numbers of pops supporting the institution
		divide = ciso_size_scaling_factor # usu. 100,000
		multiply = average_sol
		divide = 10
		round = yes
	}
}
