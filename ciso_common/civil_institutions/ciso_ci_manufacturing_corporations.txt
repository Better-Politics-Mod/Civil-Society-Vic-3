ciso_ci_manufacturing_corporations = {
	icon = "gfx/interface/icons/civil_institution_icons/manufacturing_corporations.dds"

	visible = {
		always = yes
	}

	possible = {
		custom_tooltip = {
			text = "ciso_ci_manufacturing_corporations_creation_trigger_manufacturing"
			any_scope_building = {
				level > 10
				is_building_group = bg_manufacturing
				private_ownership_fraction > 0.2
			}
		}
		custom_tooltip = {
			text = "ciso_ci_manufacturing_corporations_creation_trigger_capitalist_ratio"
			pop_type_percent_state = {
				target = pop_type:capitalists
				value >= 0.0025
			}
		}
	}

	measure_weights = {
		value = 0
		if = {
			limit = {
				scope:measure = flag:ciso_ms_schooling
			}
			add = 1
		}
	}
	
	is_radical = {
		OWNER = {
			has_law = law_type:law_industry_banned
		}
	}

	is_loyalist = {
		OWNER = {
			has_law = law_type:law_laissez_faire
		}
	}

	# stance 
	# 0 = defensive
	# >0 = aggressive
	# <0 = cooperative
	stance = {
		value = 1
	}
	
	social_impact = {
		value = organization
		multiply = size # the numbers of pops supporting the institution
		divide = ciso_size_scaling_factor # usu. 100,000
	}
}
